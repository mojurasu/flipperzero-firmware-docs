55template <class RAIt, class Compare = std::less<> >
56constexpr void quick_sort(RAIt first, RAIt last, Compare cmp = Compare{}) {
57 auto const N = cstd::distance(first, last);
58 if(N <= 1) return;
59 auto const pivot = *cstd::next(first, N / 2);
60 auto const middle1 =
61 cstd::partition(first, last, [=](auto const& elem) { return cmp(elem, pivot); });
62 auto const middle2 =
63 cstd::partition(middle1, last, [=](auto const& elem) { return !cmp(pivot, elem); });
64 quick_sort(first, middle1, cmp); // assert(std::is_sorted(first, middle1, cmp));
65 quick_sort(middle2, last, cmp); // assert(std::is_sorted(middle2, last, cmp));
66}
